LIS 和 LNDS

int LIS(vector<int>& a){
    vector<int> dp;
    int mx=0;
    for(auto it:a){
        auto pos=lower_bound(all(dp),it);
        if(pos==dp.end()) dp.pb(it);
        else *pos=it;
        mx=max(mx,(int)dp.size());
    }
    return mx;
}

int LNDS(vector<int>& a){
    vector<int> dp;
    int mx=0;
    for(auto it:a){
        auto pos=upper_bound(all(dp),it);
        if(pos==dp.end()) dp.pb(it);
        else *pos=it;
        mx=max(mx,(int)dp.size());
    }
    return mx;
}

根据两数之和和异或值反推两数

原理：a+b==(a^b)+2*(a&b)

如果$sum-xor$是奇数，那么无解。

否则$A=(sum-xor)/2$，根据A和xor的每一位填就行了，注意如果某一位两数都是1的话也是无解。
